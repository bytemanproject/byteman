#
# log a message to indicate that triggerMethod was called and fired the rule

RULE test return value binding
CLASS org.jboss.byteman.tests.misc.TestReturnBindingAssignment
METHOD triggerMethod(int)
HELPER org.jboss.byteman.tests.helpers.Default
AT EXIT
BIND test = $0;
     returnBinding = $!
IF TRUE
DO test.log("triggerMethod : triggered with " + returnBinding);
   $!= "unexpected"
ENDRULE
